#include <iostream>

using namespace std;

int main()
{
    int n;
    cin>>n;
    int arr[n];
    for(int i=0;i<n;i++){
        cin>>arr[i];
        
    }
    int xo=0;
    for(int i=0;i<n;i++){
        xo^=arr[i];
    }
    for(int i=0;i<n;i++){
        xo^=i;
        
    }
    
    int rsb=xo & -xo;
    
    int x=0;
    int y=0;
    for(auto val : arr){
        if(val & rsb ==0){
            x^=val;
        }else{
            y^=val;
        }
    }
    
    for(int i=1;i<n;i++){
        if(rsb &i==0){
            x^=i;
        }
        else{
            y^=i;
        }
    }
    
    cout<<x<<" "<<y;

    return 0;
}
